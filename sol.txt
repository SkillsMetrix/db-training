import { useState } from "react";

function Header(props) {
  // initialize the data
  const [count, setCount] = useState(10);
  const inc = () => {
    setCount(count + 1);
  };
   const dec = () => {
    setCount(count - 1);
  };
  return (
    <div>
      <p>{props.hdata}</p>

      <h3>{count}</h3>

      <hr />
      <button onClick={inc}>INC</button>
      <button onClick={dec}>DEC</button>
    </div>
  );
}

export default Header;
